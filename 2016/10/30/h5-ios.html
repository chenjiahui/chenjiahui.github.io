_bridge = [WebViewJavascriptBridge bridgeForWebView:_webView webViewDelegate:self handler:^(id data, WVJBResponseCallback responseCallback) {
        // NSLog(@"ObjC received message from JS: %@", data);
        responseCallback(@"Response for message from ObjC");
    }];
    [_bridge registerHandler:@"testObjcCallback" handler:^(id data, WVJBResponseCallback responseCallback) {
        // NSLog(@"testObjcCallback2222 called: %@", data);
        [self.navigationController popViewControllerAnimated:YES];
        responseCallback(@"Response from testObjcCallback");
    }];

android 注意：一定要开启，执行js的权限。我们当时在开发时前端没有开启，导致一直调用不成功，还以为这套方法有问题。

protected void onCteate(Bundle savedInstanceState){
    super.onCreate(savedInstanceState);
    WebView wv;
    wv = (WebView)findViewById(R.id.webView);
    wv.getSettings().setJavaScriptEnabled(true);    #开启js执行权限
    wv.loadUrl("");
    wv.addJavascriptInterface(new WebAppInterface(this), "androidItf");
}
public class WebAppInterface {
    Context mContext;

    WebAppInterface(Context c) {
        mContext = c;
    }

    @JavascriptInterface
    public void hello(String toast) {
        Toast.makeText(mContext, toast, Toast.LENGTH_SHORT).show();
    }
}

        function connectWebViewJavascriptBridge(callback) {
            if (window.WebViewJavascriptBridge) {
                callback(WebViewJavascriptBridge)
            } else {
                document.addEventListener('WebViewJavascriptBridgeReady',
                        function() {
                            callback(WebViewJavascriptBridge)
                        }, false)
            }
        }

        function testInvokIos() {
            connectWebViewJavascriptBridge(function(bridge) {
                var uniqueId = 1
                function log(message, data) {
                    var log = document.getElementById('log')
                    var el = document.createElement('div')
                    el.className = 'logLine'
                    el.innerHTML = uniqueId++ + '. ' + message
                            + (data ? ':<br/>' + JSON.stringify(data) : '')
                    if (log.children.length) {
                        log.insertBefore(el, log.children[0])
                    } else {
                        log.appendChild(el)
                    }
                }
                //初始化连接
                bridge.init(function(message, responseCallback) {
                    log('JS got a message', message)
                    alert(message)
                    var data = {
                        'type' : '2'
                    }
                    log('JS responding with', data)
                    responseCallback(data)
                })

                //注册testJavascriptHandler供ios调用
                bridge.registerHandler('testJavascriptHandler', function(data,
                        responseCallback) {
                    log('ObjC called testJavascriptHandler with', data)
                    alert(data)
                    var responseData = {
                        'Javascript Says' : 'Right back atcha!'
                    }
                    log('JS responding with', responseData)
                    responseCallback(responseData)
                })


                //向ObjC发送message
                var button = document.getElementById('buttons').appendChild(
                        document.createElement('button'))
                button.innerHTML = '向ObjC发送message'
                button.onclick = function(e) {
                    e.preventDefault()
                    var data = 'Hello from JS button'
                    log('JS sending message', data)
                    bridge.send(data, function(responseData) {
                        log('JS got response', responseData)
                    })
                }

                document.getElementById('button3').onclick = function(e) {
                    e.preventDefault()
                    log('JS calling handler "testObjcCallback"')
                    //调用ios中注册的testObjcCallback
                    bridge.callHandler('testObjcCallback', {
                        'foo' : 'bar'
                    }, function(response) {
                        log('JS got response', response)
                    })
                }

            })
        }

        function testInvokAndroid() {
            alert(window.androidItf);
            alert(window.androidItf.hello());
            var res = window.androidItfs.hello();
            alert(res);
        }
    </script>
