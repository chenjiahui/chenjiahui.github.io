    {
        if(nc.id==ncId){
            this.superTabnav.selectedChild=nc;
            var uiComponent:UIComponent = nc.getChildAt(0) as UIComponent;
            uiComponent.dispatchEvent(new Event("showRef"));//抛事件
        this.workorderlist.currentListItem=currentWorkListSelectIndex;
            this.workorderlist.dispatchEvent(new Event("listRef"));//抛事件
