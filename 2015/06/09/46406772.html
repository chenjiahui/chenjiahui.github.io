Flex_抛自定义事件

by Bug管理员
in 前端 on 2015-06-09 | tags: Flex 
public function getCount(o:Object):int{
    var sum:int = 0
    if(o.children){
        for each(var object:Object in o.children){
            sum +=(getCount(object));
        }
        return sum;
    }else{
        sum ++;
        return sum;
    }
}
