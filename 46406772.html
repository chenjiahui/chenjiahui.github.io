tags: Flex 

public function getCount(o:Object):int{
    var sum:int = 0
    if(o.children){
        for each(var object:Object in o.children){
            sum +=(getCount(object));
        }
        return sum;
    }else{
        sum ++;
        return sum;
    }
}
